""
This script generates a requirements.txt file for your Python project.

The requirements.txt file lists all the Python packages and their versions that your project depends on.
This file can be used to install the same set of dependencies on another machine, ensuring that your project
runs consistently across different environments.

The script uses the `pip freeze` command to get the list of installed packages and their versions,
and then writes this information to a file named "requirements.txt" in the current directory.
"""

import os

# Get the list of installed packages and their versions
packages = [p.strip() for p in os.popen('pip freeze').readlines()]

# Write the packages to the requirements.txt file
with open('requirements.txt', 'w') as f:
    f.write('\n'.join(packages))

print("Requirements file 'requirements.txt' generated successfully!")
